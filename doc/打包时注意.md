# 打包时需要注意的点
1. 在`罡风之息`和`龙吼功`技能定义文件里需要加上`dragonsurvival:push`效果，具体分别如下

        # 罡风之息
        {
              "effect_type": "dragonsurvival:push",
              "push_force": {
                "type": "minecraft:linear",
                "base": 0.05,
                "per_level_above_first": 0.05
              },
              "target_direction": {
                "direction": "looking_at"
              }
        }

        # 龙吼功
        {
              "effect_type": "dragonsurvival:push",
              "push_force": {
                "type": "minecraft:linear",
                "base": 0.5,
                "per_level_above_first": 0.5
              },
              "target_direction": {
                "direction": "looking_at"
              }
        }
